**「さらに読み込む」ハンドラ（per D-15）:**
```typescript
const loadMore = useCallback(async () => {
  setLoading(true)
  try {
    const res = await fetch(`/api/dashboard/timeline?offset=${days.length}&limit=20&portfolioId=${portfolioId}`)
    if (!res.ok) throw new Error('Failed to load')
    const newDays: TimelineDay[] = await res.json()
    setDays(prev => [...prev, ...newDays])
    setHasMore(newDays.length >= 20)
  } catch {
    // エラー時はsilent fail — hasMore を false にして再試行を許可しない
    setHasMore(false)
  } finally {
    setLoading(false)
  }
}, [days.length, portfolioId])
```

1. `app/api/dashboard/timeline/route.ts` を新規作成:

```typescript
import { NextResponse, type NextRequest } from 'next/server'
import { getIronSession } from 'iron-session'
import { cookies } from 'next/headers'
import { sessionOptions, type SessionData } from '@/lib/session'
import { getTimelineData, getPortfolioId } from '@/lib/dashboard/queries'

export async function GET(request: NextRequest) {
  // 認証チェック（T-04-06: proxy.ts は /api/dashboard/* をカバーするが、defense-in-depth として Route Handler 内でも確認）
  const session = await getIronSession<SessionData>(await cookies(), sessionOptions)
  if (!session.isAuthenticated) {
    return NextResponse.json({ error: 'unauthorized' }, { status: 401 })
  }

  const { searchParams } = new URL(request.url)

  // T-04-07: offset パラメータインジェクション防止 — Number() 変換 + 範囲チェック
  const rawOffset = searchParams.get('offset')
  const rawLimit = searchParams.get('limit')
  const offset = Math.max(0, Math.floor(Number(rawOffset ?? '0')) || 0)
  const limit = Math.min(100, Math.max(1, Math.floor(Number(rawLimit ?? '20')) || 20))

  // portfolioId はクエリパラメータから取得（Client Component がサーバーから受け取った値を送り返す）
  const portfolioId = searchParams.get('portfolioId')
  if (!portfolioId || !/^[0-9a-f-]{36}$/i.test(portfolioId)) {
    return NextResponse.json({ error: 'invalid portfolioId' }, { status: 400 })
  }

  try {
    const data = await getTimelineData(portfolioId, limit, offset)
    return NextResponse.json(data)
  } catch {
    return NextResponse.json(
      { error: 'データの読み込みに失敗しました' },
      { status: 500 }
    )
  }
}
```

セキュリティ対策:
- `getIronSession` で認証チェック（Spoofing 防止）
- `offset` と `limit` を `Number()` + `Math.max/min/floor` で安全に変換（Tampering 防止）
- `limit` は最大100に制限（DoS 防止）
- `portfolioId` は UUID 形式のバリデーション（Tampering 防止）
