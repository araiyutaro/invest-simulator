    import 'server-only'
    import { TICKERS } from '@/config/tickers'
    import { lastBusinessDay, isMarketClosed, resolveTargetDate } from './calendar'
    import { fetchOhlcvYahoo, fetchFxUsdJpy } from './yahoo'
    import { fetchCompanyNews, fetchBasicFinancials } from './finnhub'
    import { fetchOhlcvStooq } from './stooq'
    import { upsertPriceSnapshots, upsertNewsSnapshots, upsertFundamentalsSnapshots, writeMarketClosedRow } from './persist'
    import type { OhlcvRow, Market } from './types'
    import type { Ticker } from './types'

    export type FetchMode = 'incremental' | 'backfill'
    export type FailureSummary = {
      ok: string[]
      failed: Array<{ symbol: string; reason: string }>
      marketClosed: string[]
      durationMs: number
    }

    export type FetchOptions = {
      mode: FetchMode
      daysBack?: number              // for backfill; default 100
      now?: Date                     // for tests
      onlySymbols?: readonly string[] // CLI override
    }

    export async function fetchMarketData(opts: FetchOptions): Promise<FailureSummary> {
      const started = Date.now()
      const now = opts.now ?? new Date()
      const daysBack = opts.mode === 'backfill' ? (opts.daysBack ?? 100) : 1

      const tickers = opts.onlySymbols
        ? TICKERS.filter(t => opts.onlySymbols!.includes(t.symbol))
        : TICKERS

      const ok: string[] = []
      const failed: Array<{ symbol: string; reason: string }> = []
      const marketClosed: string[] = []

      for (const ticker of tickers) {
        try {
          const dates = resolveTargetDate(ticker.market, opts.mode, daysBack, now)
          if (dates.length === 0) {
            // All target dates are closed for this market — write one closed row for latest
            const latestIso = lastBusinessDayOrToday(ticker.market, now)
            await writeMarketClosedRow(ticker.market, latestIso, [ticker.symbol])
            marketClosed.push(ticker.symbol)
            continue
          }
          const period1 = dates[0]
          const period2 = incrementIso(dates[dates.length - 1], 1) // yahoo period2 is exclusive

          const rows = await fetchWithFallback(ticker, period1, period2)
          await upsertPriceSnapshots(rows)

          // News + fundamentals for US only, for the last date
          if (ticker.market === 'US') {
            try {
              const news = await fetchCompanyNews(ticker.symbol, period1, dates[dates.length - 1])
              await upsertNewsSnapshots(news, dates[dates.length - 1])
            } catch (e) {
              console.warn(`[market] news fetch failed for ${ticker.symbol}: ${(e as Error).message}`)
            }
            try {
              const fund = await fetchBasicFinancials(ticker.symbol)
              await upsertFundamentalsSnapshots([fund], dates[dates.length - 1])
            } catch (e) {
              console.warn(`[market] fundamentals fetch failed for ${ticker.symbol}: ${(e as Error).message}`)
            }
          }
          ok.push(ticker.symbol)
        } catch (e) {
          const reason = (e as Error).message
          console.error(`[market] ${ticker.symbol} failed: ${reason}`)
          failed.push({ symbol: ticker.symbol, reason })
        }
      }

      // FX (once per run)
      try {
        const fxDates = resolveTargetDate('US', opts.mode, daysBack, now)
        if (fxDates.length > 0) {
          const fxRows = await fetchFxUsdJpy(fxDates[0], incrementIso(fxDates[fxDates.length - 1], 1))
          await upsertPriceSnapshots(fxRows)
          ok.push('JPYUSD')
        }
      } catch (e) {
        failed.push({ symbol: 'JPYUSD', reason: (e as Error).message })
      }

      return { ok, failed, marketClosed, durationMs: Date.now() - started }
    }

    async function fetchWithFallback(ticker: Ticker, period1: string, period2: string): Promise<OhlcvRow[]> {
      // D-13 three triggers: exception, empty, stale
      const expectedLatest = period2 // the last target date (exclusive end; data should reach period2 - 1)

      if (ticker.market === 'JP') {
        // JP: yahoo primary → Stooq fallback
        try {
          const rows = await fetchOhlcvYahoo(ticker.symbol, period1, period2)
          if (rows.length === 0) throw new Error('empty yahoo response')
          if (isStale(rows, period1)) throw new Error('stale yahoo data')
          return rows
        } catch (e) {
          console.warn(`[market] yahoo failed for ${ticker.symbol}, falling back to Stooq: ${(e as Error).message}`)
          return fetchOhlcvStooq(ticker.symbol, period1, period2)
        }
      }
      // US: yahoo only for OHLCV (per Amendment A)
      return fetchOhlcvYahoo(ticker.symbol, period1, period2)
    }

    function isStale(rows: OhlcvRow[], period1: string): boolean {
      // D-13.3: latest row date < expected business day - 1. For simplicity,
      // if the max date in rows is more than 1 day before period1 window end, mark stale.
      const maxDate = rows.map(r => r.priceDate).sort().pop()
      if (!maxDate) return true
      return maxDate < period1
    }

    function incrementIso(isoDate: string, n: number): string {
      const d = new Date(`${isoDate}T00:00:00Z`)
      d.setUTCDate(d.getUTCDate() + n)
      return d.toISOString().slice(0, 10)
    }

    function lastBusinessDayOrToday(market: Market, now: Date): string {
      return lastBusinessDay(market, now)
    }

        symbol, priceDate: d, open: '1', high: '1', low: '1', close: '1', rawClose: '1', volume: '1',
        currency: symbol.endsWith('.T') ? 'JPY' : 'USD', source: 'yahoo', marketClosed: false, assetClass: 'equity',
      }))
    }

    describe('orchestrator (DATA-01..05)', () => {
      beforeEach(() => {
        vi.resetAllMocks()
        ;(persist.upsertPriceSnapshots as any).mockResolvedValue(undefined)
        ;(persist.upsertNewsSnapshots as any).mockResolvedValue(undefined)
