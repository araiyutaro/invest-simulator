    ```typescript
    #!/usr/bin/env tsx
    // CLI — run via `pnpm tsx scripts/backfill.ts [--symbol AAPL] [--days 100]`
    // Bypasses Vercel 60s timeout (PITFALLS #3) by running locally.

    import 'dotenv/config' // or whatever pattern Phase 1 uses to load .env.local
    import { fetchMarketData } from '@/lib/market/orchestrator'

    function parseArgs(argv: string[]): { symbol?: string; days: number; mode: 'incremental' | 'backfill' } {
      const out: { symbol?: string; days: number; mode: 'incremental' | 'backfill' } = { days: 100, mode: 'backfill' }
      for (let i = 0; i < argv.length; i++) {
        const a = argv[i]
        if (a === '--symbol') out.symbol = argv[++i]
        else if (a === '--days') out.days = Number(argv[++i])
        else if (a === '--mode') out.mode = argv[++i] as any
        else if (a === '--help' || a === '-h') {
          console.log('Usage: pnpm tsx scripts/backfill.ts [--symbol SYM] [--days N] [--mode incremental|backfill]')
          process.exit(0)
        }
      }
      return out
    }

    async function main() {
      const args = parseArgs(process.argv.slice(2))
      console.log(`[backfill] mode=${args.mode} days=${args.days}${args.symbol ? ` symbol=${args.symbol}` : ''}`)
      const started = Date.now()
      const result = await fetchMarketData({
        mode: args.mode,
        daysBack: args.days,
        onlySymbols: args.symbol ? [args.symbol] : undefined,
      })
      const durationS = ((Date.now() - started) / 1000).toFixed(1)
      console.log(`[backfill] completed in ${durationS}s`)
      console.log(`  ok:           ${result.ok.join(', ') || '(none)'}`)
      console.log(`  failed:       ${result.failed.map(f => `${f.symbol}:${f.reason}`).join(', ') || '(none)'}`)
      console.log(`  marketClosed: ${result.marketClosed.join(', ') || '(none)'}`)
      process.exit(result.failed.length > 0 ? 1 : 0)
    }

    main().catch((e) => {
      console.error('[backfill] fatal:', e)
      process.exit(2)
    })
    ```
