    ```typescript
    import 'server-only'
    import Anthropic from '@anthropic-ai/sdk'
    import { betaZodTool } from '@anthropic-ai/sdk/helpers/beta/zod'
    import { z } from 'zod'
    import { env } from '@/lib/env'

    export const runtime = 'nodejs'
    export const maxDuration = 60

    const client = new Anthropic({ apiKey: env.ANTHROPIC_API_KEY })

    const getPriceTool = betaZodTool({
      name: 'get_price',
      description: 'Get current stock price for a symbol',
      inputSchema: z.object({ symbol: z.string() }),
      run: async (input) => JSON.stringify({
        symbol: input.symbol, price: 150.00, currency: 'USD', date: '2026-04-11',
      }),
    })

    const placeOrderTool = betaZodTool({
      name: 'place_order',
      description: 'Place a virtual buy/sell/hold decision',
      inputSchema: z.object({
        symbol: z.string(),
        action: z.enum(['BUY', 'SELL', 'HOLD']),
        quantity: z.number().int().min(0),
        reasoning: z.string().min(1),
      }),
      run: async (input) => JSON.stringify({ success: true, ...input }),
    })

    export async function GET() {
      const startedAt = Date.now()
      try {
        const finalMessage = await client.beta.messages.toolRunner({
          model: 'claude-opus-4-5',
          max_tokens: 1024,
          system: 'You are a virtual fund manager managing a JPY 10M portfolio. Use get_price to check the current price, then use place_order to record a BUY/SELL/HOLD decision with your reasoning.',
          messages: [{ role: 'user', content: 'What should we do with AAPL today?' }],
          tools: [getPriceTool, placeOrderTool],
        })
        return Response.json({
          sdk: 'standard',
          elapsedMs: Date.now() - startedAt,
          usage: (finalMessage as any).usage ?? null,
          content: finalMessage.content,
        })
      } catch (error) {
        return Response.json({
          sdk: 'standard',
          error: error instanceof Error ? error.message : String(error),
          elapsedMs: Date.now() - startedAt,
        }, { status: 500 })
      }
    }
    ```

    **Step 3**: `app/_spikes/agent-sdk/route.ts` を作成:

    ```typescript
    import 'server-only'
    import { env } from '@/lib/env'

    export const runtime = 'nodejs'
    export const maxDuration = 60

    export async function GET() {
      const startedAt = Date.now()
      process.env.ANTHROPIC_API_KEY = env.ANTHROPIC_API_KEY
      try {
        const { query } = await import('@anthropic-ai/claude-agent-sdk')
        const messages: unknown[] = []
        for await (const message of query({
          prompt: 'You are a virtual fund manager. Decide what to do with AAPL today. Reply with a single sentence: BUY, SELL, or HOLD, with reasoning.',
          options: { allowedTools: [] },
        } as any)) {
          messages.push(message)
        }
        return Response.json({
          sdk: 'agent',
          elapsedMs: Date.now() - startedAt,
          messageCount: messages.length,
          messages,
        })
      } catch (error) {
        return Response.json({
          sdk: 'agent',
          error: error instanceof Error ? error.message : String(error),
          stack: error instanceof Error ? error.stack : undefined,
          elapsedMs: Date.now() - startedAt,
        }, { status: 500 })
      }
    }
    ```
