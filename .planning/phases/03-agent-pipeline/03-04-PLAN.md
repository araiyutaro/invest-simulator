export async function ensurePortfolio(): Promise<string> {
  const existing = await db.select({ id: portfolios.id }).from(portfolios).limit(1)
  if (existing.length > 0) return existing[0].id

  const [created] = await db.insert(portfolios).values({

}

export async function ensureMarketData(runDate: string): Promise<void> {
  const existing = await db.select({ id: priceSnapshots.id })
    .from(priceSnapshots)
    .where(eq(priceSnapshots.priceDate, runDate))
    .limit(1)
  if (existing.length > 0) return

  await fetchMarketData({ mode: 'incremental' })
}

**4. saveDecisionRecord()（D-16, AGENT-05）:**
```typescript
export async function saveDecisionRecord(params: {
  portfolioId: string
  runDate: string
  transcript: DecisionTranscript
  costUsd: number
  modelUsed: string
  summary: string
  confidence: string | null
}): Promise<{ inserted: boolean; decisionId: string | null }> {
  const result = await db.insert(decisions).values({
    portfolioId: params.portfolioId,
    runDate: params.runDate,
    transcript: params.transcript,
    tokenCostEstimate: params.costUsd.toFixed(4),
    modelUsed: params.modelUsed,
    summary: params.summary,
    confidence: params.confidence,
  }).onConflictDoNothing().returning({ id: decisions.id })

  if (result.length === 0) return { inserted: false, decisionId: null }
  return { inserted: true, decisionId: result[0].id }
}
```

```typescript
import 'server-only'
import { NextResponse, type NextRequest } from 'next/server'
import { env } from '@/lib/env'
import { GEMINI_MODEL } from '@/lib/ai/client'
import { buildSystemPrompt, buildUserPrompt } from '@/lib/agent/prompt-builder'
import { callGemini } from '@/lib/agent/gemini-caller'
import { executeDecisions } from '@/lib/agent/executor'
import {
  ensurePortfolio,
  ensureMarketData,
  loadPromptContext,
  saveDecisionRecord,
  savePortfolioSnapshot,
} from '@/lib/agent/data-loader'

export const maxDuration = 120  // D-17

export async function POST(request: NextRequest) {
  // 1. CRON_SECRET認証（既存パターン転用）
  const header = request.headers.get('authorization') ?? ''
  if (header !== `Bearer ${env.CRON_SECRET}`) {
    return NextResponse.json({ error: 'unauthorized' }, { status: 401 })
  }

  try {
    const today = new Date().toISOString().split('T')[0]  // YYYY-MM-DD

    // 2. ポートフォリオ確保（D-13）
    const portfolioId = await ensurePortfolio()

    // 3. 冪等ガード — saveDecisionRecordのINSERT ON CONFLICT DO NOTHINGで実装
    // まず試行的にプレチェック: 既に決定レコードがあるか確認
    // （実際の冪等ガードはsaveDecisionRecordで行う。ここではearly returnのための先行チェック）

    // 4. 市場データ確保（Pitfall 3対策）
    await ensureMarketData(today)

    // 5. プロンプト構築
    const promptContext = await loadPromptContext(portfolioId, today)
    const systemPrompt = buildSystemPrompt()
    const userPrompt = buildUserPrompt(promptContext)

    // 6. Gemini呼び出し（D-14リトライ含む）
    const geminiResult = await callGemini(systemPrompt, userPrompt)

    // 7. DecisionTranscript組み立て（AGENT-05）
    const transcript: DecisionTranscript = {
      system_prompt: systemPrompt,
      user_prompt: userPrompt,
      raw_messages: geminiResult.ok
        ? [
            { role: 'user', content: userPrompt },
            { role: 'model', content: geminiResult.rawText },
          ]
        : [
            { role: 'user', content: userPrompt },
            { role: 'system', content: `Error: ${geminiResult.error}` },
          ],
      input_data_snapshot: {
        as_of: new Date().toISOString(),
        universe: promptContext.tickers.map(t => t.symbol),
        prices: Object.fromEntries(promptContext.tickers.map(t => [t.symbol, { close: t.latestClose }])),
        portfolio: {
          cashJpy: promptContext.portfolio.cashJpy,
          positions: promptContext.portfolio.positions,
        },
      },
      usage: geminiResult.ok
        ? {
            input_tokens: geminiResult.usage.promptTokens,
            output_tokens: geminiResult.usage.candidateTokens,
            total_tokens: geminiResult.usage.totalTokens,
          }
        : undefined,
    }

    // 8. decisions保存（D-16 冪等INSERT）
    const saveResult = await saveDecisionRecord({
      portfolioId,
      runDate: today,
      transcript,
      costUsd: geminiResult.ok ? geminiResult.costUsd : 0,
      modelUsed: GEMINI_MODEL,
      summary: geminiResult.ok ? geminiResult.response.market_assessment.slice(0, 500) : `Error: ${geminiResult.error}`,
      confidence: null,  // 全体confidenceはdecisions配列内の個別confidenceで表現
    })

    if (!saveResult.inserted) {
      return NextResponse.json({ status: 'skipped', reason: 'already_ran_today' }, { status: 200 })
    }

    // 9. Gemini失敗時はここで終了（transcript保存済み、D-14）
    if (!geminiResult.ok) {
      return NextResponse.json({
        status: 'error',
        error: geminiResult.error,
        decisionId: saveResult.decisionId,
      }, { status: 200 })  // 200: cron retry防止のため
    }

    // 10. 売買執行
    const closePrices = new Map(promptContext.tickers.map(t => [t.symbol, t.latestClose]).filter((e): e is [string, number] => e[1] !== null))
    const currentPositions = new Map(promptContext.portfolio.positions.map(p => [p.symbol, { quantity: p.quantity, avgCost: p.avgCost, currency: p.currency }]))

    const executionResult = await executeDecisions({
      decisions: geminiResult.filteredDecisions,
      portfolioId,
      decisionId: saveResult.decisionId!,
      closePrices,
      fxRateUsdJpy: promptContext.fxRateUsdJpy,
      currentCashJpy: promptContext.portfolio.cashJpy,
      currentPositions,
    })

    // 11. portfolio_snapshot記録（D-12）
    // positionsValueJpy = 全ポジションの (close * quantity * fxRate) 合計
    let positionsValueJpy = 0
    for (const ticker of promptContext.tickers) {
      const pos = promptContext.portfolio.positions.find(p => p.symbol === ticker.symbol)
      if (!pos || pos.quantity === 0) continue
      const close = ticker.latestClose ?? 0
      const fxRate = pos.currency === 'USD' ? (promptContext.fxRateUsdJpy ?? 0) : 1
      positionsValueJpy += pos.quantity * close * fxRate
    }
    // executionResult.newCashJpy reflects post-trade cash; recalculate positionsValueJpy with updated quantities
    // For simplicity, use the execution result's newCashJpy and re-query or approximate
    await savePortfolioSnapshot({
      portfolioId,
      snapshotDate: today,
      cashJpy: executionResult.newCashJpy,
      positionsValueJpy,  // Note: This uses pre-trade positions; executor should be extended if needed
    })

    return NextResponse.json({
      status: 'success',
      decisionId: saveResult.decisionId,
      trades: executionResult.trades.length,
      skipped: executionResult.skipped.length,
      costUsd: geminiResult.costUsd,
    }, { status: 200 })

  } catch (e) {
    const msg = (e as Error).message
    console.error('[cron/daily-run] fatal:', msg)
    return NextResponse.json({ error: 'internal', message: msg }, { status: 500 })
  }
}

export async function GET() {
  return NextResponse.json({ error: 'method_not_allowed' }, { status: 405 })
}
```
